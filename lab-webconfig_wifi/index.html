ใบงานเสริม: การตั้งค่า Wi-Fi ผ่านหน้าเว็บ Web Configuration

เรียนรู้วิธีการทำงานของ Captive Portal การเขียนโปรแกรมทำหน้าเว็บต้อนรับของอุปกรณ์ IoT และการบันทึกข้อมูลแบบถาวรในระบบไฟล์ LittleFS

ข้อมูลผู้ส่งการทดลอง
โครงร่างโค้ดโปรแกรมหลัก (Fill in the Blanks)

จงเติมคำสั่งในช่องว่างให้สมบูรณ์ เพื่อให้ DNS Server ชี้ปลายทางมายังอุปกรณ์ได้ถูกต้อง และเปิดเขียนไฟล์ตั้งค่า Wi-Fi ลงใน LittleFS ได้อย่างถูกต้อง 

#include <Arduino.h>
#include <Wire.h>
#include <LittleFS.h>
#include <ArduinoJson.h>
#include <DNSServer.h>

#if defined(ESP8266)
#include <ESP8266WiFi.h>
#include <ESP8266WebServer.h>
#else
#include <WiFi.h>
#include <WebServer.h>
#endif

#define BUTTON_PIN 0

const char* apSSID = "ESP-Config-AP";
const byte DNS_PORT = 53;
IPAddress apIP(192, 168, 4, 1);

DNSServer dnsServer;
#if defined(ESP8266)
ESP8266WebServer server(80);
#else
WebServer server(80);
#endif

bool apMode = false;
String wifiSSID = "";
String wifiPassword = "";

bool loadWifiConfig() {
  if (!LittleFS.exists("/config.json")) return false;
  File file = LittleFS.open("/config.json", "r");
  if (!file) return false;
  JsonDocument doc;
  deserializeJson(doc, file);
  file.close();
  wifiSSID = doc["ssid"].as<String>();
  wifiPassword = doc["pass"].as<String>();
  return true;
}

void saveWifiConfig(String ssid, String pass) {
  // เติมคำสั่งเปิดไฟล์เพื่อเขียนข้อมูลตั้งค่าเครือข่าย Wi-Fi
  File file = ;
  
  if (file) {
    JsonDocument doc;
    doc["ssid"] = ssid;
    doc["pass"] = pass;
    serializeJson(doc, file);
    file.close();
    Serial.println("Config saved successfully.");
  }
}

void handlePortal() {
  if (LittleFS.exists("/index.html")) {
    File file = LittleFS.open("/index.html", "r");
    server.streamFile(file, "text/html");
    file.close();
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  LittleFS.begin();
  
  if (loadWifiConfig()) {
    WiFi.mode(WIFI_STA);
    WiFi.begin(wifiSSID.c_str(), wifiPassword.c_str());
    int retries = 0;
    while (WiFi.status() != WL_CONNECTED && retries < 30) {
      delay(500);
      retries++;
    }
    if (WiFi.status() == WL_CONNECTED) return;
  }
  
  // Launch Captive Portal if connection failed
  apMode = true;
  WiFi.mode(WIFI_AP);
  WiFi.softAP(apSSID);
  dnsServer.start(DNS_PORT, "*", apIP);
  server.on("/", HTTP_GET, handlePortal);
  server.begin();
}

void loop() {
  if (apMode) {
    // เติมคำสั่งประมวลผล DNS เพื่อดักจับทุกคำขอชี้ทางมายังอุปกรณ์
    ;
    
    server.handleClient();
  }
  delay(1);
}
