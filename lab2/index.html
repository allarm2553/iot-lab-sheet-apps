ใบงานที่ 2: การแสดงผลผ่านจอ OLED SSD1306 (I2C)

(SSD1306 OLED Display) - หลักสูตรการพัฒนาระบบ Hybrid Local/Cloud IoT Node ด้วย ESP32 / ESP8266

ระบบส่งใบงานปฏิบัติการออนไลน์
คู่มือใบงานการทดลอง

วัตถุประสงค์

  1. เพื่อเรียนรู้การเชื่อมต่อจอภาพ OLED ขนาดเล็กผ่านโปรโตคอลการสื่อสารแบบ I2C
  2. เพื่อแสดงข้อมูลข้อความและตัวแปรตัวเลขแบบไดนามิกบนจอภาพในพิกเซลที่ต้องการ
  3. เพื่อฝึกฝนสร้างหน้าจอแสดงสถานะอินเทอร์เฟซผู้ใช้งานแบบดั้งเดิม (Status UI)

การเชื่อมต่อวงจร (Wiring)

  • เชื่อมต่อระบบเซ็นเซอร์และรีเลย์จากใบงานที่ 1 ไว้ตามเดิม
  • SSD1306 OLED Display: จอ OLED มีการเชื่อมต่อแบบ On-board อยู่ในตัวบอร์ดแล้ว ไม่ต้องใช้สายจัมเปอร์เชื่อมต่อภายนอก:
    • IPST-WiFi (ESP32): เชื่อมต่อกับบัส I2C ที่ขา SDA = 21, SCL = 22
    • AX-WiFi (ESP8266): เชื่อมต่อกับบัส I2C ที่ขา SDA = D2 (GPIO 4), SCL = D1 (GPIO 5)

โค้ดโครงร่าง (Skeleton Code)

ศึกษาโค้ดต่อไปนี้และค้นหาคำตอบเพื่อนำมาเติมในช่องว่างทางขวามือให้สมบูรณ์:

oled_demo.ino 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

void setup() {
  Serial.begin(115200);
  #if defined(ESP8266)
  Wire.begin(4, 5); // SDA = GPIO4 (D2), SCL = GPIO5 (D1) สำหรับ ESP8266
  #else
  Wire.begin(21, 22); // SDA = GPIO21, SCL = GPIO22 สำหรับ ESP32
  #endif
  
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) { 
    Serial.println(F("SSD1306 allocation failed"));
    for(;;);
  }
  
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
}

void loop() {
  display.clearDisplay();
  
  // 1. เขียนข้อความพาดหัว (Header) ไว้ที่แถวบนสุด
  display.setCursor(0, 0);
  display.print("MONITORING NODE");
  
  // 2. วาดเส้นแบ่งกึ่งกลางหน้าจอแนวนอนที่พิกเซล Y = 10
  display.drawLine(0, 10, 128, 10, [ ช่องที่ 1 ] );
  
  // 3. แสดงค่าเซ็นเซอร์อุณหภูมิที่อ่านได้ (สมมติค่าอุณหภูมิ = 28.5)
  display.setCursor(0, 15);
  display.printf("Temp: %.1f C", 28.5);
  
  // 4. แสดงผลข้อความส่งคำสั่งออกจอภาพจริง
  display. [ ช่องที่ 2 ] ;
  delay(1000);
}

โจทย์ท้าทายการทดลอง (Challenge)

โจทย์: ให้นักเรียนนำค่าเซ็นเซอร์ทั้งหมดจากใบงานที่ 1 (อุณหภูมิ, ความชื้น และค่าเปอร์เซ็นต์แอนะล็อก) ร่วมกับสัญญานอินพุตปุ่มกดและสถานะการทำงานของรีเลย์จากใบงานที่ 1.1 (เช่น แสดงค่าอุณหภูมิ ความชื้น แอนะล็อก และแสดงสถานะ Relay: ON/OFF พร้อมจำนวนครั้งที่กดปุ่ม Toggle ในบรรทัดล่างสุด) มาแสดงผลบนหน้าจอ OLED ร่วมกันอย่างถูกต้องและสวยงาม
