ใบงานที่ 3.1: การเชื่อมต่อ WiFi เบื้องต้น

(WiFi Scan & Station Mode) — หลักสูตรการพัฒนาระบบ Hybrid Local/Cloud IoT Node ด้วย ESP32 / ESP8266

ระบบส่งใบงานปฏิบัติการออนไลน์
คู่มือใบงานการทดลอง

วัตถุประสงค์

  1. เพื่อเรียนรู้การใช้งาน WiFi Library บน ESP32 / ESP8266 สำหรับสแกนหาเครือข่ายไร้สายในบริเวณใกล้เคียง
  2. เพื่อเข้าใจกระบวนการเชื่อมต่อ WiFi ในโหมด Station (STA) และตรวจสอบสถานะการเชื่อมต่อ
  3. เพื่อสามารถอ่านค่า IP Address ที่ได้รับจาก DHCP Server และแสดงผลผ่าน Serial Monitor

อุปกรณ์ที่ต้องใช้

รายการอุปกรณ์ รายละเอียด จำนวน
บอร์ดไมโครคอนโทรลเลอร์ ESP32 DevKitC หรือ ESP8266 NodeMCU / IPST-WiFi / AX-WiFi 1 บอร์ด
สาย USB USB-to-Serial (Micro-USB หรือ USB-C ตามรุ่นบอร์ด) 1 เส้น
คอมพิวเตอร์ + Arduino IDE ติดตั้ง ESP32 / ESP8266 Board Package เรียบร้อยแล้ว 1 ชุด
เครือข่าย WiFi Access Point ที่รู้จัก SSID และ Password เพื่อใช้เชื่อมต่อ 1 เครือข่าย

การทดลอง

ทดลองที่ 1 — สแกนหาเครือข่าย WiFi

ในการทดลองนี้ บอร์ดจะทำการสแกนหาเครือข่าย WiFi ทั้งหมดที่อยู่ในบริเวณใกล้เคียง แล้วแสดงรายการ SSID, ช่องสัญญาณ (Channel) และความแรงของสัญญาณ (RSSI) ผ่าน Serial Monitor

โค้ดโครงร่าง (Skeleton Code) — WiFi Scan

ศึกษาโค้ดต่อไปนี้แล้วค้นหาคำตอบเพื่อนำมาเติมในช่องว่างทางขวามือ:

wifi_scan.ino
Arduino C++ 
#if defined(ESP8266)
  #include <ESP8266WiFi.h>
#elif defined(ESP32)
  #include <WiFi.h>
#endif

void setup() {
  Serial.begin(115200);
  Serial.println("\n=== WiFi Scanner ===");

  // 1. ตั้งค่าโหมด WiFi เป็น Station (STA) Mode
  WiFi.mode( [ ช่องที่ 1 ] );

  Serial.println("กำลังสแกนหาเครือข่าย...");
}

void loop() {
  // 2. เริ่มสแกนเครือข่าย และรับจำนวนเครือข่ายที่พบ
  int n = WiFi.scanNetworks();

  if (n == 0) {
    Serial.println("ไม่พบเครือข่าย WiFi");
  } else {
    Serial.printf("\nพบเครือข่ายทั้งหมด: %d เครือข่าย\n", n);
    Serial.println("-----------------------------------------------");
    Serial.printf("%-4s %-30s %-8s %-6s %s\n",
                  "#", "SSID", "CH", "RSSI", "Security");
    Serial.println("-----------------------------------------------");

    for (int i = 0; i < n; i++) {
      // 3. อ่านชื่อ SSID ของเครือข่ายลำดับที่ i
      String ssid = WiFi.[ ช่องที่ 2 ](i);

      // 4. อ่านความแรงสัญญาณ RSSI (dBm)
      int rssi = WiFi.[ ช่องที่ 3 ](i);

      // 5. อ่านหมายเลขช่องสัญญาณ (Channel)
      int ch = WiFi.[ ช่องที่ 4 ](i);

      // 6. ตรวจสอบประเภทความปลอดภัย (ถ้า OPEN ให้แสดง "OPEN" ไม่งั้นแสดง "WPA/WPA2")
      String security = (WiFi.encryptionType(i) == [ ช่องที่ 5 ]) ? "OPEN" : "WPA/WPA2";

      Serial.printf("%-4d %-30s %-8d %-6d %s\n",
                    i + 1, ssid.c_str(), ch, rssi, security.c_str());
    }
    Serial.println("-----------------------------------------------");
  }

  // ล้างผลการสแกนและรอ 10 วินาทีก่อนสแกนใหม่
  WiFi.scanDelete();
  delay(10000);
}

หมายเหตุเรื่อง encryptionType

ค่าคงที่ที่ใช้ตรวจสอบประเภทการเข้ารหัส WiFi แตกต่างกันระหว่างชิป:

  • ESP32: ใช้ WIFI_AUTH_OPEN เพื่อตรวจสอบเครือข่ายแบบเปิด
  • ESP8266: ใช้ ENC_TYPE_NONE เพื่อตรวจสอบเครือข่ายแบบเปิด

โดยในโค้ดนี้ช่องที่ 5 คือชื่อค่าคงที่ที่ถูกต้องสำหรับแพลตฟอร์มที่ใช้งาน

ทดลองที่ 2 — เชื่อมต่อ WiFi (Station Mode)

ในการทดลองนี้ บอร์ดจะเชื่อมต่อ WiFi ในโหมด Station (STA) โดยระบุ SSID และ Password ของ Access Point ที่ต้องการ จากนั้นรอจนกว่าการเชื่อมต่อสำเร็จ และแสดง IP Address ที่ได้รับ

โค้ดโครงร่าง (Skeleton Code) — WiFi Connect

ศึกษาโค้ดต่อไปนี้แล้วค้นหาคำตอบเพื่อนำมาเติมในช่องว่างทางขวามือ:

wifi_connect.ino
Arduino C++ 
#if defined(ESP8266)
  #include <ESP8266WiFi.h>
#elif defined(ESP32)
  #include <WiFi.h>
#endif

// กำหนด SSID และ Password ของ Access Point ที่ต้องการเชื่อมต่อ
const char* ssid     = "iot_512";
const char* password = "iot123456";

void setup() {
  Serial.begin(115200);
  Serial.println("\n=== WiFi Station Mode ===");

  // 1. เริ่มต้นการเชื่อมต่อ WiFi ด้วย SSID และ Password
  WiFi.[ ช่องที่ 6 ](ssid, password);
  Serial.print("กำลังเชื่อมต่อกับ ");
  Serial.print(ssid);

  // 2. รอจนกว่าสถานะ WiFi จะเท่ากับ WL_CONNECTED (เชื่อมต่อสำเร็จ)
  while (WiFi.status() != [ ช่องที่ 7 ]) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\n✓ เชื่อมต่อ WiFi สำเร็จ!");
  Serial.print("SSID     : ");
  Serial.println(WiFi.SSID());

  // 3. แสดง IP Address ที่ได้รับจาก DHCP Server
  Serial.print("IP Address : ");
  Serial.println(WiFi.[ ช่องที่ 8 ]());

  // 4. แสดงความแรงของสัญญาณ RSSI (dBm)
  Serial.print("RSSI     : ");
  Serial.print(WiFi.[ ช่องที่ 9 ]());
  Serial.println(" dBm");

  // 5. แสดง MAC Address ของบอร์ด
  Serial.print("MAC Addr : ");
  Serial.println(WiFi.[ ช่องที่ 10 ]());
}

void loop() {
  // ตรวจสอบสถานะการเชื่อมต่อทุก 5 วินาที
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("[OK] WiFi ยังคงเชื่อมต่ออยู่ | IP: " + WiFi.localIP().toString());
  } else {
    Serial.println("[!!] WiFi หลุดการเชื่อมต่อ! กำลังพยายามเชื่อมต่อใหม่...");
    WiFi.reconnect();
  }
  delay(5000);
}
